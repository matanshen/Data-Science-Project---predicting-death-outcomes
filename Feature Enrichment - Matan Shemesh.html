def obesity_status(value):
    if value < 18.5:
        return "underweight"
    if 18.5 <= value < 25:
        return "healthy"
    elif 25 <= value < 30:
        return "overweight"
    else:
        return "obesity"

def calculate_egfr(age, sex, creatinin):

  if sex == 1:
    egfr = 141 * (creatinin ** (-0.411)) * (age ** (-0.203))
  elif sex == 2:
    egfr = 141 * (creatinin ** (-0.411)) * (age ** (-0.107))

  return round(egfr,3)

df['eGFR'] = df.apply(lambda x: calculate_egfr(age = x['age'], sex = x['sex'], creatinin = x['creatinin']), axis=1)

def calculate_risk(age, sex, albumin, platelets):

  risk = 0

  if age >= 20 and age <= 39:
    risk += 5
  elif age >= 40 and age <= 59:
    risk += 10
  elif age >= 60:
    risk += 20

  if sex == 1:
    risk += 10
  elif sex == 2:
    risk += 5

  if albumin < 3.5:
    risk += 10
  elif albumin >= 3.5 and albumin < 4.5:
    risk += 5
  elif albumin >= 4.5:
    risk += 0

  if platelets < 150:
    risk += 0
  elif platelets >= 150 and platelets < 400:
    risk += 5
  elif platelets >= 400:
    risk += 10

  if risk < 10:
    return "Low"
  elif risk >= 10 and risk < 20:
    return "Moderate"
  elif risk >= 20 and risk < 30:
    return "High"
  elif risk >= 30:
    return "Very high"
