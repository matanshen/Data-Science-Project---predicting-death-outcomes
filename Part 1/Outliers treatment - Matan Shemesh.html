def create_outlier_mat(df):
    mat = np.zeros(df.shape)
    
    for col_idx, col in enumerate(df.columns):
        if col in num_vars:
            s = df[col]
            q1 = s.quantile(0.25)
            q3 = s.quantile(0.75)
            iqr = q3-q1
            lower = q1 - 1.5*iqr
            upper = q3 + 1.5*iqr

            outliers = s[(s<=lower) | (s>=upper)]
        
            for row_idx in outliers.index:
                mat[row_idx, col_idx] = 1
    
    return mat

outlier_mat = create_outlier_mat(df)

def cocor(x1,y1, x2,y2):
    xy1 = x1.corr(y1, method='spearman')
    xy2 = x2.corr(y2, method='spearman')
    n1 = len(x1)
    n2 = len(x2)
    xy_z = 0.5 * np.log((1 + xy1)/(1 - xy1))
    ab_z = 0.5 * np.log((1 + xy2)/(1 - xy2))
    if n2 is None:
        n2 = n1
    se_diff_r = np.sqrt(1/(n1 - 3) + 1/(n2 - 3))
    diff = xy_z - ab_z
    z = abs(diff / se_diff_r)
    p = (1 - norm.cdf(z)) * 2
    return z, p
