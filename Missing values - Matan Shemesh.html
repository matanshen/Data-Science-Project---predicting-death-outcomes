def convert_var_to_categorial(df, var):
    
    if df[var].dtype.name == 'category':
        df[var] = df[var].cat.add_categories('unknown')
        for i, val in enumerate(df[var]):
            if np.isnan(val):
                df.loc[i, var] = 'unknown'
    else:
        s = df[var]
        converted_to_cat = []
        
        for i, val in enumerate(df[var]):
            if np.isnan(val):
                converted_to_cat.append('unknown')
            elif val < s.quantile(0.25):
                converted_to_cat.append(1)
            elif val < s.quantile(0.5):
                converted_to_cat.append(2)
            elif val < s.quantile(0.75):
                converted_to_cat.append(3)
            else:
                converted_to_cat.append(4)
        #change the type of varibales to catgory and replace them to categorial vars
        var_to_cat = pd.Series(converted_to_cat, dtype='category')
        df[var] = var_to_cat
        # add the variable to categorial vars
        cat_vars.append(var)
        #remove the variable from numerical vars
        num_vars.remove(var)
    return df
